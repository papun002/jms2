View for All product list-------------------

CREATE VIEW `view_product_jms` AS select product_jms.client_id AS client_id, category_jms.category_id AS category_id, product_jms.product_id AS product_id, category_jms.category_name AS category_name, product_jms.product_name AS product_name, product_jms.product_weight AS product_weight, product_jms.product_huid_no AS product_huid_no, product_jms.product_hsn_no AS product_hsn_no, product_jms.product_desc as product_desc, product_jms.product_barcode AS product_barcode, product_jms.status AS status FROM (product_jms JOIN category_jms ON (product_jms.category_id = category_jms.category_id));

View for All client-------------------

CREATE VIEW `view_all_clients` AS select client_sub_jms.admin_id AS admin_id, client_sub_jms.client_id AS client_id, client_jms.client_name AS client_name, client_jms.contact_person AS contact_person, client_jms.client_contact AS client_contact, client_jms.client_add AS client_add, client_jms.client_email AS client_email, client_jms.client_gst AS client_gst, client_jms.client_user AS client_user, client_jms.client_password AS client_password, client_sub_jms.plan_id AS plan_id, plan_jms.plan_name As plan_name, plan_jms.plan_duration AS plan_duration, plan_jms.plan_price AS plan_price, client_sub_jms.sub_id AS sub_id, client_sub_jms.start_date AS start_date, client_sub_jms.end_date AS end_date, client_sub_jms.sub_price AS sub_price, client_sub_jms.sub_status AS sub_status FROM ((client_sub_jms JOIN client_jms ON (client_sub_jms.client_id = client_jms.client_id)) JOIN plan_jms ON (client_sub_jms.plan_id = plan_jms.plan_id));
